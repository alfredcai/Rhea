$(function () {

function CtrlViewQueryCourseTable() {
var istudentNo = $.trim($("#StudentNo").val());
if (istudentNo == "") {
return;
}
$.ajax({
type: "POST",
url: "/StudentQuery/CtrlViewQueryCourseTableTab",
data: { studentNo: istudentNo },
cache: false,
beforeSend: function () {
$("#divCourseTable").html("<div style=\"text-align:center; margin-top:10px;\"><img src='/Content/default/images/loading.gif' alt=''/></div>");
},
success: function (html) {
$("#divCourseTable").empty().html(html);
}
});
}
//提交到 非校区提示
function SubmitForm() {
var count = 0;
for (var i = 0; i < 6; i++) {
if ($.trim($("#CID" + i).val()) != ""
&& $.trim($("#TNo" + i).val()) != "") {
count++;
}
}
//禁用 添加按钮
DisabledInput("FastInputAction");
if (Number(count) > 0) {
//处理 checkbox 的值
HandleCheckValue();
//设置 非本校区提示界面 关闭时 恢复添加按钮
$("#divNotCollge").dialog("option", { width: 900, height: 500, close: function () { DisabledInputNot("FastInputAction"); } });
$("#divNotCollge").dialog("open");
$("#postForm").ajaxSubmit({
url: "/CourseSelectionStudent/CtrlViewNotCollegeMessage",
type: "POST",
beforeSend: function () {
$("#divNotCollge").html("<div style=\"text-align:center; margin-top:10px;\"><img src='/Content/default/images/loading.gif' alt=''/></div>");
},
success: function (data) {
if (data.toString().indexOf("没有非本校区课程") != -1) {
//直接提交结果
SubmitFormHandle();
} else {
//显示 非本校区课程提示
$("#divNotCollge").empty().html(data);
}
}
});
} else {
//提示 并 恢复 添加按钮
alert("请最少输入一门课程！");
DisabledInputNot("FastInputAction");
}
}
//提交到 处理选课
function SubmitFormHandle() {
//关闭 非本校区提示
CloseDialog("divNotCollge");
//结果界面设置 关闭时 刷新 课表 添加按钮恢复
$("#divOperationResult").dialog("option", { width: 900, height: 500, close: function () { CtrlViewQueryCourseTable(); DisabledInputNot("FastInputAction"); } });
$("#divOperationResult").dialog("open");
$("#postForm").ajaxSubmit({
url: "/CourseSelectionStudent/CtrlViewOperationResult",
type: "POST",
beforeSend: function () {
$("#divOperationResult").html("<div style=\"text-align:center; margin-top:10px;\"><img src='/Content/default/images/loading.gif' alt=''/></div>");
},
success: function (html) {
$("#divOperationResult").empty().html(html);
}
});
}
//处理复选框
function HandleCheckValue() {
$(".checkBook").val("0");
$(".checkBook:checked").each(function () {
$(this).val("1");
});
}
